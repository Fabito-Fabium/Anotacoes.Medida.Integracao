\textbf{Sejam $(X, \mcal F, \mu)$ um espaĂ§o mensurĂĄvel e $A \in \mcal F$. 
Mostre que a aplicaĂ§ĂŁo $\lambda: \mcal F \rightarrow \xbR$ dada por 
$\lambda(E)\coloneq \mu(E \cap A)$ ĂŠ uma medida.}
\begin{proof}[\textbf{Dem:}]
Ora, sabendo que $\mu$ ĂŠ uma medida, temos 
$\lambda(\emptyset) = \mu(\emptyset \cap A) = \mu(\emptyset) = 0$. Mais ainda, 
visto que para todo $B \in \mathcal F$, $\mu(B) \ge 0$, temos que dado $E \in \mathcal F$,
$E \cap A \in \mcal F$, ou seja, 
$$\lambda(E) = \mu(E \cap A) \ge 0.$$ 
Para verificar a propriedade de $\sigma$-aditividade, considere ${E_n}$ uma famĂ­lia em 
$\mathcal F$, dois a dois disjuntos. Como consequĂŞncia da definiĂ§ĂŁo de ${E_n}$, temos que 
$\{E_n \cap A\}$ tambĂŠm ĂŠ dois a dois disjuntos. Com isso,
\begin{align*}
   \lambda\left(\bigcup\limits_{n \in \mathbb N} E_n\right) &= 
   \mu\left(\left(\bigcup\limits_{n \in \mathbb N} E_n\right)\cap A\right) = 
   \mu\left(\bigcup\limits_{n \in \mathbb N} (E_n \cap A)\right),\\
   &=\sum_{n \in \mathbb N}\mu(E_n \cap A)=\sum_{n\in\mbb N}\lambda(E_n).
\end{align*}
Desse modo, segue que $\lambda$ ĂŠ uma medida, como querĂ­amos.
\end{proof}

\textbf{Sejam $(X, \mcal F)$ um espaĂ§o mensurĂĄvel $\mu_1, \cdots, \mu_n$ medidas em 
$(X, \mcal F)$ e $a_1, \cdots, a_n$ reais nĂŁo negativos. Mostre que a aplicaĂ§ĂŁo 
$\lambda: \mcal F \rightarrow \xbR$ dada por 
\begin{align*}
  \lambda(E) = \sum_{k=1}^{n} a_k\mu_k(E)
\end{align*}
ĂŠ uma medida em $(X, \mcal F)$.}
\begin{proof}[\textbf{Dem:}]
Visto que para todo $k \in \mbb N_{\le n}$, temos que $\mu_k$ ĂŠ uma medida em $(X, \mathcal F)$,
entĂŁo, para cada uma das medidas, segue que:
\begin{enumerate}
  \item $\mu_k(\emptyset) = 0$; 
  \item $\mu_k(E) \ge 0, \text{ para todo } E \in \mcal F$; 
  \item Vale a $\sigma$-aditividade para sequĂŞncias de conjuntos disjuntos.
\end{enumerate}
Desse modo,
\begin{align*}
  \lambda(\emptyset) = \sum_{k=1}^{n} a_k\mu_k(\emptyset) \stackrel{(1)}{=} 
  \sum_{k=1}^{n} a_k \cdot0 = \sum_{k=1}^n 0 = 0. 
\end{align*}
Mais ainda, visto que $a_1, \dots, a_n$ ĂŠ uma sequĂŞncia de nĂşmeros nĂŁo negativos,
  temos por $(2)$, que $a_k \mu_k \ge 0$ para todo $k \in \mbb N_{\le n}$. Ou seja 
\begin{align*}
  \lambda(E) = \sum_{k=1}^n a_k \mu_k(E) \ge 0, \text{ para todo } E \in \mcal F.
\end{align*}
  Por fim, dada uma sequĂŞncia $\{E_j\}_{j \in \mbb N}$ em $\mcal F$ disjunto dois a dois,
  temos, 
  \begin{align*}
    \lambda\left(\bigcup\limits_{j\in \mbb N} E_j \right) &= 
    \sum_{k=1}^{n} \left[a_k \mu_k\left(\bigcup\limits_{j\in \mbb N} E_j\right)\right]
    \stackrel{(3)}{=}\sum_{k=1}^{n} \left[a_k \sum_{j \in \mbb N}\mu_k(E_j)\right]\\ 
    &= \sum_{k=1}^{n} \left[\sum_{j \in \mbb N}a_k\mu_k(E_j)\right]
    \end{align*}
Como a soma em $k$ ĂŠ finita, podemos trocar a ordem das somas, obtendo
\begin{align*}
    &= \sum_{j \in \mbb N} \left[\sum_{k \in \mbb N_{\le n}}a_k\mu_k(E_j)\right]
    = \sum_{j \in \mbb N} \lambda(E_j).
  \end{align*}
  Com isso, temos que $\lambda$ ĂŠ uma medida em $(X, \mcal F)$.
\end{proof}

\textbf{
  Sejam $(X, \mcal F)$ um espaĂ§o mensurĂĄvel e $(\mu_n)_{n \in \mbb N}$ uma sequĂŞncia 
  de medidas em $(X, \mcal F)$ tal que $\mu_k(X) = 1$, para todo $k \in \mbb N$. 
  Mostre que a aplicaĂ§ĂŁo $\lambda: \mcal F \rightarrow \xbR$ dada por 
\begin{align*}
  \lambda(E) = \sum_{k \in \mbb N} \frac{\mu_k(E)}{2^k},
\end{align*}
ĂŠ uma medida e $\lambda(X) = 1$.
}
\begin{proof}[\textbf{Dem:}] Vejamos que $\lambda(\emptyset) = 0$. 
Ora, $\lambda(\emptyset) = \sum_{k \in \mbb N} \frac{\mu_k(\emptyset)}{2^k}$. JĂĄ que 
  $\mu_k(\emptyset) = 0$ para todo $k \in \mbb N$, temos $\lambda(\emptyset) = 0$.
  Daqui, visto que para cada $E \in \mcal F$ e $k \in \mbb N$ temos $\mu_k(E)/2^k \ge 0$, 
entĂŁo, para todo $E \in \mcal F$,
  \begin{align*}
    \lambda(E) = \sum_{k \in \mbb N} \frac{\mu_k(E)}{2^k} \ge 0.
  \end{align*}
  Com isso, vejamos que $\lambda$ ĂŠ $\sigma$-aditiva. Dada uma sequĂŞncia 
  $\{E_j\}_{j \in \mbb N}$ em $\mcal F$ disjunto dois a dois, temos, 
  \begin{align*}
    \lambda\left(\bigcup E_j\right) = 
    \sum_{k \in \mbb N} \frac{\mu_k \left(\bigcup E_j\right)}{2^k}.
  \end{align*}
  JĂĄ que vale a $\sigma$-aditividade para todo $\mu_k$, temos,
  \begin{align*}
    \lambda\left(\bigcup E_j\right) = 
    \sum_{k \in \mbb N}\left(\sum_{j \in \mbb N} \frac{\mu_k (E_j)}{2^k} \right).
  \end{align*}
  Como $\mu_k(E_j)/2^k \ge 0$ para cada $j,k \in \mbb N$,
 \begin{align*}
    \lambda\left(\bigcup E_j\right) = 
    \sum_{j \in \mbb N}\left(\sum_{k \in \mbb N} \frac{\mu_k (E_j)}{2^k} \right) = 
    \sum_{j \in \mbb N}\lambda(E_j).
  \end{align*}
 Por fim, visto que $\mu_k(X) = 1$ para todo $k \in \mbb N$, 
\begin{align*}
  \lambda(X) = \sum_{k \in \mbb N} \frac{\mu_k(X)}{2^k} 
  = \sum_{k \in \mbb N} \frac{1}{2^k} = 1. 
\end{align*}
Com isso, temos o que querĂ­amos.
\end{proof}

\textbf{Seja $(\mbb N, \mcal P(\mbb N))$ e $(a_n)_{n \in \mbb N}$ uma sequĂŞncia 
de nĂşmeros reais estendidos nĂŁo negativos. Defina $\mu: \mcal P(\mbb N) \rightarrow \xbR$
por $\mu(\emptyset) = 0$ e, para $E \neq \emptyset$,
\begin{align*}
  \mu(E) = \sum_{n \in E} a_n.
\end{align*}
Mostre que $\mu$ ĂŠ uma medida em $\mbb N$. Reciprocamente, mostre que toda medida 
em $(\mbb N, \mcal P(\mbb N))$ ĂŠ obtida desta maneira, para alguma sequĂŞncia 
$(a_n)$ em $\xbR$.
}
\begin{proof}[\textbf{Dem:}]
  Pela definiĂ§ĂŁo de $\mu$, temos $\mu(\emptyset) = 0$. Mais ainda, como $a_n \ge 0$ para todo $n \in \mbb N$, dado 
  $E \in \mcal P(\mbb N)$, temos
  $\mu(E) = \sum_{n \in E} a_n \ge 0$. Daqui, vejamos que $\mu$ ĂŠ $\sigma$-aditiva. Dada uma sequĂŞncia 
  $\{E_n\}_{n \in \mbb N}$ em $\mcal P(\mbb N)$ disjuto dois a dois. Se $\bigcup E_j = \emptyset$, entĂŁo $E_j = \emptyset$ para todo 
  $j \in \mbb N$. Portanto, $\mu(\bigcup E_j) = 0 = \sum \mu(E_j)$. Assumimos entĂŁo, que $\bigcup E_j \neq \emptyset$, 
  desse modo,
  \begin{align*}
    \mu\left(\bigcup\limits_{j \in \mbb N} E_j\right) &= \sum_{n \in \bigcup E_j} a_n.
  \end{align*}
  Considere $J = \{n \in \mbb N\st  E_n = \emptyset\}$. Visto que $\{E_j\}$ ĂŠ disjunto dois a dois,
  para todo $n \in \bigcup E_j$ existe um Ăşnico $k \in \mathbb N$ tal que 
  $n \in E_k$, e, como para cada $n \in \bigcup E_j$, temos $a_n \ge 0$, segue que,
  \begin{align*}
    &= \sum_{n \in \bigcup E_j} a_n = \sum_{j \in \mbb N \setminus J} \left[\sum_{n \in E_j} a_n\right]
    = \sum_{j \in \mbb N \setminus J} \mu(E_j).
  \end{align*}
  Da definiĂ§ĂŁo de $J$, 
\begin{align*}
  &= \sum_{j \in \mbb N \setminus J} \mu(E_j) + \sum_{l \in J}\mu(E_l) = \sum_{j \in \mbb N} \mu(E_j).
\end{align*}
  Com isso, segue que $\mu$ ĂŠ $\sigma$-aditiva. JĂĄ que $\mcal P (\mbb N)$ ĂŠ uma $\sigma$-ĂĄlgebra, $\mu$ ĂŠ medida em 
  $(\mbb N, \mcal P (\mbb N))$.

  Agora, vejamos que toda medida em $\mbb N$ pode ser escrita como $\mu$. Considere $\lambda$ 
  uma medida em $(\mbb N, \mcal P (\mbb N))$. Seja $E \in \mcal P (\mbb N)$ tal que $E \neq \emptyset$.
  EntĂŁo, 
  \begin{align*}
    \lambda(E) = \lambda\left(\bigcup\limits_{n \in E} \{n\}\right).
  \end{align*}
  Como $E \subseteq \mbb N$, $E$ ĂŠ enumerĂĄvel. Mais ainda, $\{n\}_{n \in E}$ ĂŠ disjunto dois a dois, portanto, 
  da $\sigma$-aditividade, 
  \begin{align*}
    &= \lambda\left(\bigcup\limits_{n \in E} \{n\}\right) = \sum_{n \in E} \lambda(\{n\}),
  \end{align*}
  definindo $a_n = \lambda(\{n\})$, temos que cada $a_n \in \xbR$ ĂŠ nĂŁo-negativo, 
  como querĂ­amos.
\end{proof}

\textbf{Seja $X$ um conjunto nĂŁo-enumerĂĄvel. Considere a coleĂ§ĂŁo 
$$\mcal F = \{A \subseteq X\st  A \text{ ou } A^C \text{ ĂŠ enumerĂĄvel}\}.$$ Defina $\mu: \mcal F\rightarrow \mbb R$
por 
\begin{align*}
  \mu(E) = \begin{cases}
    0, &\text{ se $E$ ĂŠ enumerĂĄvel;}\\
    1, &\text{ se $E^C$ ĂŠ enumerĂĄvel.}
  \end{cases}
\end{align*}
Mostre que $\mu$ ĂŠ uma medida em $(X, \mcal F)$.
}
\begin{proof}[\textbf{Dem: }]
  Primeiro, note que $\mcal F$ de fato ĂŠ uma $\sigma$-ĂĄlgebra. E, visto 
  que $\emptyset$ ĂŠ enumerĂĄvel, $\mu(\emptyset) = 0$, e como para todo $E \in \mcal F$, temos $\mu(E) = 0$ 
  ou $\mu(E) = 1$, segue que $\mu(E) \ge 0$. Daqui, vejamos que $\mu$ ĂŠ $\sigma$-aditiva. 
  Considere a sequĂŞncia $\{E_j\}_{j \in \mbb N}$ em $\mcal F$ disjunto dois a dois. Se $\bigcup E_j$ ĂŠ enumerĂĄvel, 
  entĂŁo $E_j$ ĂŠ enumerĂĄvel para todo $j \in \mbb N$, desse modo, 
  \begin{align*}
    \mu\left(\bigcup\limits_{j \in \mbb N} E_j\right) = 0 = \sum_{j \in \mbb N} \mu(E_j)
  \end{align*}
  Caso contrĂĄrio, se $\bigcup E_j$ ĂŠ nĂŁo-enumerĂĄvel, note que existe exatamente um $k \in \mbb N$ tal que $E_k$ ĂŠ 
  nĂŁo-enumerĂĄvel, pois, supondo por contradiĂ§ĂŁo que existe $m \in \mbb N$ tal que $E_m$ ĂŠ nĂŁo-enumerĂĄvel e $k \neq m$. 
  Como $E_k$ ĂŠ nĂŁo-enumerĂĄvel e $E_k \in \mcal F$, entĂŁo $E_k^C$ ĂŠ enumerĂĄvel. Entretanto, jĂĄ que a sequĂŞncia $\{E_j\}$
  ĂŠ disjunto dois a dois, $E_m \subset E_k^C$, ou seja, $E_m$ deve ser enumerĂĄvel, uma contradiĂ§ĂŁo. Com isso, garantimos 
  que existe exatamente um $k \in \mbb N$ tal que $E_k$ ĂŠ nĂŁo-enumerĂĄvel. Daqui, como $\bigcup E_j \in \mcal F$ ĂŠ nĂŁo enumerĂĄvel,
  pela definiĂ§ĂŁo de $\mcal F$, segue que 
  $\left(\bigcup E_j\right)^C$ ĂŠ enumerĂĄvel. Desse modo, visto que $\mu(E_j) = 0$ para todo 
  inteiro positivo $j \neq k$, $\mu(E_k) = 1$, temos
  \begin{align*}
    \mu\left(\bigcup\limits_{j \in \mbb N} E_j\right) = 1 = \mu(E_k) + \sum_{j \neq k} \mu(E_j) 
    = \sum_{j \in \mbb N} \mu(E_j).
  \end{align*}
  Ou seja, $\mu$ ĂŠ $\sigma$-aditiva. Portanto, $\mu$ ĂŠ uma medida em $(X, \mcal F)$.
\end{proof}

\textbf{Seja $X$ um conjunto nĂŁo-enumerĂĄvel e considere o espaĂ§o mensurĂĄvel $(X, \mcal P(X))$. 
Mostre que a aplicaĂ§ĂŁo $\mu: \mcal P(X) \rightarrow \xbR$ definida por 
\begin{align*}
  \mu(E) = \begin{cases}
    0, &\text{se $E$ ĂŠ enumerĂĄvel;}\\
    +\infty, &\text{se $E$ ĂŠ nĂŁo-enumerĂĄvel.}
  \end{cases}
\end{align*}
Mostre que $\mu$ ĂŠ uma medida em $(X, \mcal F)$.}
\begin{proof}[\textbf{Dem:}]
Visto que $\emptyset$ ĂŠ enumerĂĄvel, $\mu(\emptyset) = 0$. JĂĄ que para todo $A \in \mcal P(X)$, 
  $A$ ĂŠ enumerĂĄvel ou nĂŁo-enumerĂĄvel, temos que $\mu(A) = 0$ ou $\mu(A) = +\infty$, ou seja, 
  $\mu(A) \ge 0$. Daqui, vejamos que $\mu$ ĂŠ $\sigma$-aditiva. Considere uma sequĂŞncia 
  $\{E_j\}_{j \in \mbb N}$ em $\mcal F$ disjunto dois a dois. Se $\bigcup E_j$ ĂŠ enumerĂĄvel, 
  para todo $k \in \mbb N$, $E_k$ ĂŠ enumerĂĄvel. Sendo assim, 
  \begin{align*}
    \mu\left(\bigcup\limits_{j \in \mbb N} E_j\right) = 0

  \end{align*}
  Caso contrĂĄrio, se $\bigcup E_j$ ĂŠ nĂŁo-enumerĂĄvel, existe ao menos um $k \in \mbb N$ 
  tal que $E_k$ ĂŠ nĂŁo-enumerĂĄvel. Portanto, 
  \begin{align*}
    \mu\left(\bigcup_{j \in \mbb N} E_j\right) = +\infty = \mu(E_k)
    = \sum_{j \in \mbb N} \mu(E_j).
  \end{align*}
  Com isso, segue que $\mu$ ĂŠ $\sigma$-aditiva. Mais ainda, $\mu$ ĂŠ uma medida.
\end{proof}

  \end{align*}
  Com isso, temos que $\mu$ ĂŠ uma medida.
\end{proof}

\textbf{Sejam $(X, \mcal F, \mu)$ um espaĂ§o de medida e $(A_n)_{n \in \mbb N}$
uma sequĂŞncia em $\mcal F$.}
\begin{enumerate}[label=(\alph*), wide, labelindent=0pt]
  \item \textbf{Mostre que o conjunto 
    $$\lim \sup A_n \coloneq \bigcap_{m=1}^\infty\left
    [\bigcup\limits_{n = m}^\infty A_n\right]$$ 
    consiste dos elementos de $X$ que pertencem a infinitos conjuntos da sequĂŞncia $A_n$.}
    \begin{proof}[\textbf{Dem:}] Para verificar a proposiĂ§ĂŁo, dividiremos a demonstraĂ§ĂŁo 
      em duas partes. \\
      ($\implies$) Suponha $x \in \lim\sup A_n$, vejamos que $x$ pertence a infinitos 
      conjuntos da sequĂŞncia $A_n$. Da hipĂłtese, temos $x \in \bigcap_{m=1}^\infty 
      \left[\bigcup_{n = m}^\infty A_n\right]$, ou seja, $x \in \bigcup_{n=m}^\infty A_n$
      para todo $m \in \mbb N$. portanto, para todo $m \in \mbb N$, existe $n \ge m$ tal que 
      $x \in A_n$, i.e, $x$ pertence a infinitos conjuntos de $\{A_n\}$.\\
      ($\impliedby$) Suponhamos agora que $x$ pertence a infinitos conjuntos da sequĂŞncia 
      $A_n$ e vejamos que $x \in \lim\sup A_n$. Ora, considere a sequĂŞncia 
      $\{B_n\}_{n \in \mbb N}$ tal que $B_m = \bigcup_{n=m}^\infty A_n$. Da hipĂłtese, 
      segue que $x \in B_n$ para todo $n \in \mbb N$, ou seja, 
      $x \in \bigcap\limits_{m=1}^\infty B_m = \lim\sup A_n$. 
    \end{proof}
  \item \textbf{Se $\mu\left(\bigcup\limits_{n \in \mbb N} A_n\right) < +\infty$, mostre que 
    $$\lim \sup \mu (A_n) \le \mu(\lim\sup A_n).$$
  DĂŞ um exemplo em que a desigualdade ĂŠ falsa quando $\mu\left(\bigcup
    \limits_{n \in \mbb N} A_n\right) = +\infty$.}
    \begin{proof}[\textbf{Dem:}] Definindo $F_m = \bigcup\limits_{n = m}^\infty A_n$, temos 
      $F_{m + 1} \subseteq F_{m}$, ou seja, $\{F_m\}$ ĂŠ decrescente e mais ainda, 
        $\mu(F_1) = \mu\left(\bigcup\limits_{n \in \mbb N} A_n\right) < +\infty$. 
        Portanto, da continuidade da medida, 
        $$\mu\left(\bigcap_{m \in \mbb N} F_m\right) =
        \mu\left(\bigcap_{m=1}^\infty\left[\bigcup\limits_{n = m}^\infty A_n\right]\right)
        = \mu\left(\lim\sup A_n\right) = \lim_{m \rightarrow +\infty} \mu(F_m).$$
      Visto que cada $A_m \subseteq F_m$, temos $\mu(F_m) \ge \mu(A_m)$. Portanto, 
      \begin{align*}
        \mu(\lim\sup A_n) = \lim_{n \rightarrow +\infty}\mu(F_n) = 
        \lim\sup \mu(F_n) \ge \lim\sup \mu(A_n).
      \end{align*}
      Com isso, temos o que querĂ­amos. Como exemplo em que a desigualdade ĂŠ falsa, 
      considere $A_n = (n, +\infty)$. JĂĄ que $\mu(A_n) = +\infty$ para todo $n \in \mbb N$,
      entĂŁo, $\lim\sup \mu(A_n) = +\infty$. Entretanto, visto que para todo $x \in \mbb R$,
      existe $M \in \mbb N$ tal que $x < M$, segue que $\lim\sup A_n = \emptyset$, ou seja, 
      $\mu(\lim\sup A_n) = 0$.
    \end{proof}
  \item \textbf{Mostre que o conjunto 
    $$\lim \inf A_n \coloneq \bigcup\limits_{m=1}^\infty\left[\bigcap\limits_{n=m}^\infty 
    A_n\right]$$
    consiste dos elementos de $X$ que nĂŁo pertencem a apenas finitos conjuntos da  
    sequĂŞncia $A_n$.}
    \begin{proof}[\textbf{Dem:}] Para verificar a proposiĂ§ĂŁo, dividiremos a demonstraĂ§ĂŁo 
      em duas partes.\\
      ($\implies$) Suponhamos que $x \in \lim\inf A_n$, vejamos que $x$ nĂŁo pertence a apenas 
      finitos conjuntos da sequĂŞncia $A_n$, i.e, existe $m \in \mbb N$ tal que para 
      inteiros $n \ge m$, temos $x \in A_n$. Ora, jĂĄ que
      $$x \in \lim\inf A_n = \bigcup\limits_{m=1}^\infty\left[\bigcap\limits_{n=m}^\infty 
    A_n\right],$$
      segue que existe $m \in \mbb N$ tal que para inteiros $n \ge m$, temos $x \in A_n$,
      como querĂ­amos.\\
      ($\impliedby$) Suponhamos que $x$ nĂŁo pertence a apenas finitos conjuntos da sequĂŞncia 
      $\{A_n\}$, vejamos que $x \in \lim\inf A_n$. Ora, da hipĂłtese, temos que existe 
      $m \in \mbb N$ tal que para inteiros $n \ge m$ temos $x \in A_n$. Desse modo, 
      $$x \in \bigcup\limits_{m \in \mbb N}\left[\bigcap\limits_{n=m}^\infty A_n\right]
      = \lim \inf A_n$$
      como querĂ­amos.
    \end{proof}
  \item \textbf{Mostre que $\mu(\lim \inf A_n) \le \lim\inf \mu(A_n)$.}
    \begin{proof}[\textbf{Dem:}] Definindo $F_m = \bigcap\limits_{n = m}^\infty A_n$,
      note que $F_m \subset F_{m+1}$, sendo assim, $\{F_m\}_{m \in \mbb N}$ ĂŠ crescente. 
      Da continuidade da medida, segue que 
      \begin{align*}
        \mu\left(\bigcup_{m \in \mbb N} F_m\right)
        = \mu(\lim\inf A_n)
        = \lim_{m \rightarrow +\infty} \mu(F_m) = \lim\inf \mu(F_n).
      \end{align*}
      JĂĄ que $F_n \subset A_n$, temos $\mu(F_n) \le \mu(A_n)$, ou seja,
      \begin{align*}
        \mu(\lim\inf A_n) \le \lim\inf \mu(A_n),
      \end{align*}
      como querĂ­amos.
    \end{proof}
\end{enumerate}

\textbf{Prove o \textit{Lema de Borel-Cantelli}: Sejam $(X, \mcal F, \mu)$ um espaĂ§o 
de medida e $(E_n)_{n \in \mbb N}$ uma sequĂŞncia em $\mcal F$ tal que 
$\sum_{n \in \mbb N} \mu(E_n) < +\infty$. EntĂŁo quase-$\mu$ todo ponto $x \in X$ pertence a, 
no mĂĄximo, um nĂşmero finito de conjuntos da sequĂŞncia $(E_n)_{n \in \mbb N}$}
\begin{proof}[\textbf{Dem:}] Note que, $x\in X$ pertence a, no mĂĄximo, um nĂşmero 
  finito de conjuntos da sequĂŞncia $(E_n)_{n \in \mbb N}$ $\mu$-q.t.p se 
  $\mu(\lim\sup E_n) = 0$. Daqui, visto que 
  \begin{align*}
    \mu\left(\bigcup\limits_{n \in \mbb N}
    E_n\right) \le \sum_{n \in \mbb N} \mu(E_n) < +\infty,
  \end{align*}
  definindo $F_m = \bigcup_{n = m}^\infty E_n$, 
  segue que $\{F_m\}$ ĂŠ decrescente, ou seja, 
  pela continuidade da medida que $\lim \mu (F_n) = \mu(\bigcap F_m) = \mu(\lim\sup E_n)$. 
  Mais ainda, definindo $S_m = \sum_{n = m}^\infty \mu(E_n)$, como $\sum \mu(E_n) < +\infty$,
  segue que $\lim S_m = 0$. Portanto, da subaditividade, 
  \begin{align*}
    \mu(F_n) = \mu\left(\bigcup\limits_{n = m}^\infty E_n\right)
    \le \sum_{n = m}^\infty \mu(E_n) = S_m,
  \end{align*}
  ou seja, $\lim S_m = \lim \mu\left(\bigcup\limits_{n = m}^\infty E_n\right) 
  = \mu(\lim\sup E_n) = 0$.
\end{proof}
