\textbf{Seja $X$ um conjunto nĂŁo-enumerĂĄvel. Considere a seguinte coleĂ§ĂŁo:
$$\mathcal F = 
\{A \subseteq X ; A \text{ ou } A^C \text{ ĂŠ enumerĂĄvel}\}.$$ 
  Mostre que $\mathcal F$ ĂŠ uma $\sigma$-ĂĄlgebra.}
\\

\begin{proof}[\textbf{Dem:}]
 Ora, visto que $\emptyset$ ĂŠ enumerĂĄvel, $\emptyset \in \mathcal F$. 
E, jĂĄ que $\emptyset = X^C$, temos $X \in \mathcal F$. Mais ainda,
para qualquer $A \in \mathcal F$ temos que $A$ ou $A^C$ ĂŠ enumerĂĄvel. 
Em ambos os casos, podemos concluĂ­r que $A^C \in \mathcal F$. Desse modo, $\mathcal F$ 
satisfaz as duas primeiras condiĂ§Ăľes de uma $\sigma$-ĂĄlgebra de $X$.

Para verificar a terceira condiĂ§ĂŁo, seja 
$\{A_n\}_{n \in \mathbb N} \subset \mathcal F$. Note que 
podemos recair a dois casos, sĂŁo eles:
\begin{enumerate}
  \item Existe $k \in \mathbb N$ tal que $A_k^C$ ĂŠ enumerĂĄvel;
  \item Para todo $n \in \mathbb N$ temos $A_n^C$ nĂŁo-enumerĂĄvel. 
\end{enumerate}

Assumindo (1), jĂĄ que subconjuntos de conjuntos enumerĂĄveis sĂŁo 
enumerĂĄveis, segue que $\bigcap A_n^C \subset A_k^C$ ĂŠ enumerĂĄvel, ou seja 
$\left(\bigcap A_n^C\right)^C = \bigcup A_n \in \mathcal F$. 

Caso contrĂĄrio, assumindo (2), jĂĄ que para todo $n \in \mathbb N$, 
temos $A_n^C$ nĂŁo-enumerĂĄvel e $A_n \in \mathcal F$, entĂŁo todo $A_n$ ĂŠ enumerĂĄvel.
Desse modo, visto que a uniĂŁo enumerĂĄvel de enumerĂĄveis ĂŠ enumerĂĄvel, temos 
$\bigcup A_n$ enumerĂĄvel, ou seja $\bigcup A_n \in \mathcal F$. Portanto 
$\mathcal F$ satisfaz as condiĂ§Ăľes de uma $\sigma$-ĂĄlgebra de $X$, 
como querĂ­amos. 
\end{proof}

\textbf{Caracterize a $\sigma$-ĂĄlgebra de Borel de $\mathbb R$.}
\noindent\textbf{} 
\begin{proof}[\textbf{Dem:}]
Considere os seguintes conjuntos:
\begin{align*}
  &E_1 = \{(a, b) ; a<b\}, &&E_2= \{[a, b]; a< b\},\\
  &E_3 = \{(a, b] ; a<b\}, &&E_4= \{[a, b); a< b\},\\
  &E_5 = \{(a, +\infty) ; a\in\mathbb R\}, &&E_6= \{(+\infty, a); a \in \mathbb R\}, \\
  &E_7 = \{[a, +\infty) ; a\in\mathbb R\}, &&E_8= \{(+\infty, a]; a \in \mathbb R\}.
\end{align*}
Vejamos que para cada $n \in \mathbb N_{\leq 8}$, $\mathcal B(\mathbb R, \tau)$
ĂŠ gerada por $E_n$.

(i) ($\subseteq$) Note que $E_1 \subseteq \tau \subset \sigma(\tau)$, 
desse modo $\sigma(E_1) \subseteq \sigma(\tau)$.\\ 

(i)($\supseteq$) Por outro lado, visto que todo elemento de $\tau$ ĂŠ um aberto
e, todo aberto pode ser escrito como a uniĂŁo enumerĂĄvel de elementos de $E_1$, 
segue que $\tau \subseteq \sigma(E_1)$. Com isso, temos $\sigma(\tau) \subseteq \sigma(E_1)$.
Dito isso, segue que $E_1$ gera $\sigma(\tau)=\mathcal B (\mathbb R)$. 

(ii) ($\subseteq$) Ora, para cada $[a, b] \in E_2$, 
$$[a, b] = \bigcap\limits_{n \in \mathbb N} (a - 1/n, b + 1/n),$$
portanto $E_2 \subseteq \sigma(E_1)$, isto ĂŠ, $\sigma(E_2) \subseteq \sigma(E_1)$. 

(ii) ($\supseteq$) Por outro lado, para cada $(a, b) \in E_1$,
$$(a, b) = \bigcup\limits_{n \ge 3} [a + (b-a)/n, b - (b-a)/n],$$
desse modo, $E_1 \subseteq \sigma(E_2)$, ou seja, $\sigma(E_1) \subseteq \sigma(E_2)$.\\
Sendo assim, $\sigma(E_1) = \sigma(E_2)$, isto ĂŠ, $E_2$ gera $\mathcal B(\mathbb R)$.

(iii) e (iv) JĂĄ que cada elemento de $E_1$, $E_3$ e $E_4$ 
podem ser rescritos das seguintes formas: 
\begin{align*}
  &(a, b] = \bigcap\limits_{n \in \mathbb N} (a, b+1/n), 
  &&(a, b) = \bigcup\limits_{n \ge 2} (a, b-(b-a)/n],\\
  &[a, b) = \bigcap\limits_{n \in \mathbb N} (a-1/n, b), 
  &&(a, b) = \bigcup\limits_{n \ge 2} [a+(b-a)/n, b).
\end{align*}
Segue de argumentos similares a (ii) que $E_3$ e $E_4$ geram $\mathcal B(\mathbb R, \tau)$.

  (v), (vi), (vii) e (viii) Note que $\sigma(E_5) = \sigma(E_8)$, $\sigma(E_6) = \sigma(E_7)$ e 
  $\sigma(E_5) = \sigma(E_7)$ por argumentos semelhantes a (i). Daqui, visto que 
\begin{align*}
  &(a, +\infty) = \bigcup\limits_{n \in \mathbb N} (a, b+n), 
  &(a, b) = (a, +\infty) \cap \left(\bigcup\limits_{n \in \mathbb N} (b-1/n, +\infty)^C \right)
\end{align*}
  temos $\sigma(E_1) = \sigma(E_5)$. Com isso, para todo $n \in \mathbb N_{\le 8}$, $E_n$ gera 
  $\mathcal B(\mathbb R, \tau)$, como querĂ­amos.
\end{proof}

\textbf{Para $E \in \mathcal B(\mathbb R)$, considere $E_1 = E \cup \{+\infty\}$, 
$E_2 = E \cup \{-\infty\}$ e $E_3 = E \cup \{-\infty, +\infty\}$. Prove que 
$\{E, E_1, E_2, E_3; E\in \mathcal B(\mathbb R)\}$ 
ĂŠ uma $\sigma$-ĂĄlgebra de $\overline{\mathbb R}$}. 
\begin{proof}[\textbf{Dem:}] Defina $\mathcal C = \{E, E_1, E_2, E_3; E
\in \mathcal B(\mathbb R)\}$,
visto que $\emptyset\in \mathcal B(\mathbb R)$, segue que $\emptyset \in \mathcal C$.
De maneira anĂĄloga, temos $\overline{\mathbb R} \in \mathcal C$. Agora, seja $A \in C$, 
\begin{itemize}
  \item Para $A \not\supseteq \{-\infty, +\infty\}$, temos $A \in \mathcal B(\mathbb R)$, 
    desse modo, $\mathbb R\setminus A \in \mathcal B(\mathbb R)$, sendo assim,
    $\left(\mathbb R \setminus A \right)\cup \{-\infty, +\infty\} = A^C \in \mathcal C$.

  \item Para $A \supset \{+ \infty\}$ sem conter $\{- \infty\}$,
    note que $A^C = X \cup \{- \infty\}$ com 
    $X \in \mathcal B (\mathbb R)$, sendo assim, $A^C \in E_2 \subset \mathcal C$. 
  \item Para $A \supset \{- \infty\}$ sem conter $\{+ \infty\}$,
    segue de maneira anĂĄloga ao 
    caso anterior que $A^C \in \mathcal C$. 
  \item Para $A \supset \{- \infty, + \infty\}$, temos $A^C \in \mathcal B\{\mathbb R\}$, 
    portanto, $A^C \in \mathcal C$
\end{itemize}
Para verificar que $\mathcal C$ obedesce a terceira propriedade de $\sigma$-ĂĄlgebra,
note que todo $A \in \mathcal C$ 
temos $A = B \cup D$, onde $B \in \mathcal B (\mathbb R)$ e 
$D \in \mathbb P (\{-\infty, +\infty\}) $. Portanto, dada qualquer 
sequĂŞncia $\{A_k\}_{k \in \mathbb N}
\subset \mathcal C$, temos $\bigcup A_k = \bigcup \left(B_k \cup D_k\right)$ para algum 
$\{B_k\} \subset \mathcal B (\mathbb R)$ e 
  $\{D_k\}\subset \mathbb P (\{-\infty, +\infty\})$. Daqui, jĂĄ que 
\begin{align*}
  \bigcup A_k &= \bigcup \left(B_k \cup D_k\right),\\
  &= \left(\bigcup B_m\right) \cup \left(\bigcup D_n\right), 
\end{align*}
e, $\bigcup D_n \in \mathbb P (\{-\infty, +\infty\}) \subset \mathcal C$, assim como 
  $\bigcup B_m \in \mathcal B(\mathbb R) \subset \mathcal C$, entĂŁo 
  $\bigcup A_k \in \mathcal C$.
\end{proof}

\textbf{Sejam $f \in M(X, \mathcal F)$ e $n \in \mathbb N$. Considere o truncamento de $f$, 
$f_n: X \rightarrow \mathbb R$ definido por
\begin{align*}
  f_n(x) =\begin{cases}
    -n, &\text{se } f(x) < -n,\\
    f(x), &\text{se } |f(x)| \le n,\\
    n, &\text{se } f(x)>n.
  \end{cases}
\end{align*}
Mostre que $f_n$ ĂŠ mensurĂĄvel para todo $n \in \mathbb N$ e que $\lim f_n(x) = f(x)$, 
para todo $x \in X$.}
\begin{proof}[\textbf{Dem:}] Fixado $n \in \mathbb N$, seja $\alpha \in \mathbb R$. 

\begin{itemize}
  \item Para $\alpha < -n$, temos $\{x \in X; f_n(x) > \alpha\}
  = X \in \mathcal F$;
  \item   Para $-n \le \alpha < n$, $f_n = f$, sendo assim, $\{x \in X; f_n(x) > \alpha\} = 
    \{x \in X; f(x) > \alpha\}$. Visto que $f$ ĂŠ mensurĂĄvel, 
    $\{x \in X; f_n(x) > \alpha\} \in \mathcal F$;
  \item Por fim, para $\alpha \ge n$, $\{x \in X; f(x) > \alpha\} = \emptyset \in \mathcal F$.
\end{itemize}
Portanto, para todo $n \in \mathbb N$ temos $f_n$ mensurĂĄvel. 

Vejamos que $\lim f_n(x) = f(x)$.
Dado $x \in X$, como $f(x) \in \mathbb R$, existe $M \in \mathbb N$ tal que $-M\le f(x)\le M$ 
(e.g o menor inteiro maior que o valor absoluto de $f(x)$).
Desse modo, para inteiros $n \ge M$, temos $f_n(x) = f(x)$, 
ou seja, para todo $\epsilon > 0$, $|f_n(x) - f(x)| = 0 < \epsilon$, 
isto ĂŠ, $\lim f_n (x) = f(x)$, como querĂ­amos.
\end{proof}

\textbf{Mostre que a soma, o produto, o valor absoluto, e limites de funĂ§Ăľes mensurĂĄveis
a valores complexos sĂŁo mensurĂĄveis.}
\begin{proof}[\textbf{Dem:}] Por definiĂ§ĂŁo, uma funĂ§ĂŁo a valores complexos ĂŠ mensurĂĄvel se, 
  e somente se, a parte real e a parte imaginĂĄria da funĂ§ĂŁo sĂŁo funĂ§Ăľes mensurĂĄveis. 
  Desse modo, sejam $f, g: X \rightarrow \mathbb C$ funĂ§Ăľes mensurĂĄveis, tais que 
  $f_1, f_2, g_1, g_2: X \rightarrow \mathbb R$ satisfazem $f(x) = f_1(x)+if_2(x)$
  e $g(x)=g_1(x)+ig_2(x)$ para todo $x \in X$.
  \vspace{5mm}

\noindent
(Soma:) Ora,
\begin{align*}
  (f+g)(x) &= f(x) + g(x) = [f_1(x)+if_2(x)] + [g_1(x)+ig_2(x)],\\
  &= [f_1(x) + g_1(x)] + i[f_2(x) + g_2(x)].
\end{align*}
Visto que $f_1(x)$ e $g_1(x)$ sĂŁo funĂ§Ăľes mensurĂĄveis, temos $(f_1 + g_1)(x)$ mensurĂĄvel. Analogmante temos 
  $(f_2+g_2)(x)$ mensurĂĄvel. JĂĄ que a parte real e a parte imaginĂĄria de $(f+g)(x)$ ĂŠ mensurĂĄvel, temos 
  $(f+g)(x)$ mensurĂĄvel, como querĂ­amos.
  \vspace{5mm}

\noindent
  (Produto:) JĂĄ que, 
\begin{align*}
  (f\cdot g)(x) &= f(x)\cdot g(x) = [f_1(x)+if_2(x)]\cdot[g_1(x) + ig_2(x)], \\
  &= [(f_1\cdot g_1)(x) - (f_2\cdot g_2)(x)] + i[(f_1\cdot g_2)(x) + (f_2 \cdot g_1)(x)],\\
\end{align*}
  e temos $f_n,g_n: X \rightarrow \mathbb R$ mensurĂĄveis para $n=1, 2$, segue que $(f_i\cdot g_j)(x)$ ĂŠ 
  mensurĂĄvel para todo par $i, j \in \{1, 2\}$. Sendo assim, $(f_1\cdot g_1)(x) - (f_2\cdot g_2)(x)$ e 
  $(f_1\cdot g_2)(x) + (f_2 \cdot g_1)(x)$ sĂŁo tambĂŠm mensurĂĄveis, ou seja $(f\cdot g)(x)$ ĂŠ mensurĂĄvel.
\end{proof}

\textbf{Seja $f: X \rightarrow Y$ uma funĂ§ĂŁo. Mostre que se $\mathcal G$ 
for uma $\sigma$-ĂĄlgebra de Y, entĂŁo 
$$\mathcal F= \{f^{-1}(E);E\in\mathcal G\} $$
ĂŠ uma $\sigma$-ĂĄlgebra de X.
\begin{proof}[\textbf{Dem:}]
Ora, como $\emptyset \in \mathcal G$, entĂŁo $f^{-1}(\emptyset) = \emptyset \in \mathcal F$. E, visto que 
  $Y \in \mathcal G$, entĂŁo $f^{-1}(Y) = X \in \mathcal F$. Daqui, note que para todo $A \in \mathcal F$, 
  existe $B \in \mathcal G$ tal que $f^{-1}(B) = A$. Como $\mathcal G$ ĂŠ $\sigma$-ĂĄlgebra de $Y$, temos 
  $B^C \in \mathcal G$, sendo assim, 
  $$f^{-1}(B^C) = \left(f^{-1}(B)\right)^C = A^C \in \mathcal F.$$
  Por fim, 
  note que para toda sequĂŞncia $\{A_n\}_{n \in \mathbb N} \subset \mathcal F$, existe uma sequĂŞncia 
  $\{B_n\}_{n \in \mathbb N}$ tal que $f^{-1}(B_n) = A_n$ para todo $n 
  \in \mathbb N$. Como $\bigcup B_n \in \mathcal G$
  \begin{align*}
    f^{-1}\left(\bigcup\limits_{n \in \mathbb N} B_n\right) = \bigcup\limits_{n \in \mathbb N} 
    f^{-1}(B_n) = \bigcup\limits_{n \in \mathbb N} A_n \in \mathcal F.
  \end{align*}
Com isso, temos que $\mathcal F$ ĂŠ uma $\sigma$-ĂĄlgebra de $X$, como querĂ­amos. 
\end{proof}

\mysubsec{ExercĂ­cio 7}
\bigline
\textbf{Seja $f:X \rightarrow Y$ uma funĂ§ĂŁo. Mostre que se $\mathcal F$ 
for uma $\sigma$-ĂĄlgebra de $X$, entĂŁo 
$$\mathcal G = \{E \subset Y; f^{-1}(E)\in \mathcal F\}$$ 
ĂŠ uma $\sigma$-ĂĄlgebra de Y.
}
\begin{proof}[\textbf{Dem:}]
  Ora, como $X \in \mathcal F$ e $f^{-1}(Y) = X$, entĂŁo $Y \in \mathcal G$. Mais ainda, 
  $\emptyset \in \mathcal F$ e $f^{-1}(\emptyset) = \emptyset$, portanto $\emptyset \in \mathcal G$.

  Daqui, note que para todo $A \in \mathcal G$, existe $B \in \mathcal F$ tal que $B = f^{-1}(A)$. 
  Visto que $\mathcal F$ ĂŠ uma $\sigma$-ĂĄlgebra de $X$, entĂŁo $B^C \in \mathcal F$. JĂĄ que, 
  $$B^C = (f^{-1}(A))^C = f^{-1}\left(A^C\right) \in \mathcal F,$$
  temos $A^C \in \mathcal G$. Por fim, note que para toda sequĂŞncia $\{A_n\}_{n\in \mathbb N} \subset \mathcal G$,
  existe uma sequĂŞncia $\{B_n\}_{n \in \mathbb N} \subset \mathcal F$, tal que
  $B_n = f^{-1}(A_n)$ para todo $n \in \mathbb N$. Visto que $\mathcal F$ ĂŠ uma $\sigma$-ĂĄlgebra de $X$, temos 
  $\bigcup B_n \in \mathcal F$, isto ĂŠ,
  \begin{align*}
    \bigcup\limits_{n \in \mathbb N} B_n = \bigcup\limits_{n \in \mathbb N} f^{-1}(A_n) = 
    f^{-1}\left(\bigcup\limits_{n \in \mathbb N} A_n\right) \in \mathcal F,
  \end{align*}
 ou seja, $\bigcup A_n \in \mathcal G$. Com isso, segue que $\mathcal G$ ĂŠ uma $\sigma$-ĂĄlgebra de $Y$.
\end{proof}

\textbf{Seja $(X, \mathcal F)$ um espaĂ§o mensurĂĄvel e $f: X \rightarrow Y$ uma funĂ§ĂŁo. Seja $\mathcal C$ uma coleĂ§ĂŁo de 
subconjuntos de $Y$ tais que $f^{-1}(A) \in \mathcal F$, para todo $A \in \mathcal C$. Mostre que $f^{-1}(E) \in \mathcal F$, 
para todo $E \in \sigma(\mathcal C)$}.
\begin{proof}[\textbf{Dem:}]
  Visto que $\mathcal C = \{E\subseteq Y; f^{-1}(E) \in \mathcal F\}$, onde 
  $\mathcal F$ ĂŠ uma $\sigma$-ĂĄlgebra de $X$, segue do exercĂ­cio 
  anterior que $\mathcal C$ ĂŠ uma $\sigma$-ĂĄlgebra de $Y$. Daqui, como 
  $\mathcal C \subseteq \sigma(\mathcal C)$ e toda $\sigma$-ĂĄlgebra que contenha $\mathcal C$
  contĂŠm $\sigma(\mathcal C)$, temos que $\sigma(\mathcal C) \subseteq \mathcal C$, ou seja, 
  $\sigma(\mathcal C) = \mathcal C$. Portanto, da definiĂ§ĂŁo de $\mathcal C$, 
  segue que para todo 
  $E \in \mathcal C =  \sigma(\mathcal C)$, temos $f^{-1}(E) \in \mathcal F$, 
  como querĂ­amos.
\end{proof}

\textbf{Sejam $(X, \mathcal F)$ um espaĂ§o mensurĂĄvel e $f: X \rightarrow \mathbb R$ uma 
funĂ§ĂŁo. Mostre que $f$ ĂŠ mensurĂĄvel se, e somente se, $f^{-1}(E) \in \mathcal F$, 
para todo $E \in \mathcal B(\mathbb R)$. Conclua que as definiĂ§Ăľes de mensurabilidade
da funĂ§ĂŁo $f$ vistas em aula coincidem ao considerar $Y=(\mathbb R, \mathcal B(\mathbb R))$.
}
\begin{proof}[\textbf{Dem:}]Seja  
  $\mathcal C=\{(\alpha, +\infty)\subset \mathbb R; \alpha \in \mathbb R\}$ uma coleĂ§ĂŁo de 
  subconjuntos de $\mathbb R$. 

  \vspace{3mm}
($\implies$) Suponhamos $f$ mensurĂĄvel, isto ĂŠ,
  para todo $\alpha \in \mathbb R$, 
  $A_{\alpha} = \{x \in X; f(x) > \alpha\} \in \mathcal F$.  
   Visto que para todo $\alpha \in \mathbb R$, segue que 
  $A_\alpha = f^{-1}((\alpha, +\infty)) \in \mathcal F$, temos do exercĂ­cio anterior que
  $f^{-1}(E) \in \mathcal F$, para todo $E \in \sigma(\mathcal C)$. JĂĄ que 
  $\mathcal C$ gera $\mathcal B(\mathbb R)$, entĂŁo 
  $\sigma(\mathcal C) = \mathcal B (\mathbb R)$, como querĂ­amos.

  
  ($\impliedby$) Por outro lado, suponhamos que $f^{-1}(E) \in \mathcal F$ para todo $E \in 
  \mathcal B(\mathbb R)$. Como $\mathcal C$ gera $\mathcal B(\mathbb R)$, temos 
  $\mathcal C \subset \mathcal B(\mathbb R)$. Desse modo, para todo $E \in \mathcal C$, 
  segue da hipĂłtese que $f^{-1}(E) \in \mathcal F$. Daqui, note que para todo 
  $E \in \mathcal C$, existe $\alpha \in \mathbb R$ tal que $E = (\alpha, +\infty)$, 
  desse modo, 
  \begin{align*}
    f^{-1}(E) = f^{-1}((\alpha, +\infty)) = \{x\in X; f^{-1}(x) > \alpha\}.
  \end{align*}
  Portanto, para todo $\alpha \in \mathbb R$, temos 
  $\{x \in X; f^{-1}(x)>\alpha \} \in \mathcal F$, i.e, $f$ ĂŠ mensurĂĄvel, como querĂ­amos.

  Mais ainda, por definiĂ§ĂŁo, dizemos que $f: (X,\mathcal F) \rightarrow (Y, \mathcal G)$ ĂŠ
  mensurĂĄvel se $f^{-1}(E) \in \mathcal F$, para todo $E \in \mathcal G$. De fato, ao tomar 
  $(Y, \mathcal G)$ como sendo $(\mathbb R, \mathcal B(\mathbb R))$, temos a primeira parte
  da proposiĂ§ĂŁo do presente exercĂ­cio.
\end{proof}
